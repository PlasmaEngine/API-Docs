185 virtual plUInt64 Skip(plUInt64 uiBytes)
186 {
187 plUInt8 uiTempBuffer[1024];
188
189 plUInt64 uiBytesSkipped = 0;
190
191 while (uiBytesSkipped < uiBytes)
192 {
193 plUInt64 uiBytesToRead = plMath::Min<plUInt64>(uiBytes - uiBytesSkipped, 1024);
194
195 plUInt64 uiBytesRead = Read(uiTempBuffer, uiBytesToRead);
196
197 uiBytesSkipped += uiBytesRead;
198
199 // Terminate early if the stream didn't read as many bytes as we requested (EOF for example)
200 if (uiBytesRead < uiBytesToRead)
201 break;
202 }
203
204 return uiBytesSkipped;
205 }
